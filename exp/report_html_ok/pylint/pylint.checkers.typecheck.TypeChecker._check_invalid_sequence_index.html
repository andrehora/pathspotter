7            parent_type, (nodes.ClassDef, astroid.Instance)
8        ) or not has_known_bases(parent_type):
9            return None
10
11        # Determine what method on the parent this index will use
12        # The parent of this node will be a Subscript, and the parent of that
13        # node determines if the Subscript is a get, set, or delete operation.
14        if subscript.ctx is astroid.Store:
15            methodname = "__setitem__"
16        elif subscript.ctx is astroid.Del:
17            methodname = "__delitem__"
18        else:
19            methodname = "__getitem__"
20
21        # Check if this instance's __getitem__, __setitem__, or __delitem__, as
22        # appropriate to the statement, is implemented in a builtin sequence
23        # type. This way we catch subclasses of sequence types but skip classes
24        # that override __getitem__ and which may allow non-integer indices.
25        try:
26            methods = astroid.interpreter.dunder_lookup.lookup(parent_type, methodname)
27            if methods is astroid.Uninferable:
28                return None
29            itemmethod = methods[0]
30        except (
31            astroid.AttributeInferenceError,
32            IndexError,
33        ):
34            return None
35        if (
36            not isinstance(itemmethod, nodes.FunctionDef)
37            or itemmethod.root().name != "builtins"
38            or not itemmethod.parent
39            or itemmethod.parent.frame().name not in SEQUENCE_TYPES
40        ):
41            return None
42
43        index_type = safe_infer(subscript.slice)
44        if index_type is None or index_type is astroid.Uninferable:
45            return None
46        # Constants must be of type int
47        if isinstance(index_type, nodes.Const):
48            if isinstance(index_type.value, int):
49                return None
50        # Instance values must be int, slice, or have an __index__ method
51        elif isinstance(index_type, astroid.Instance):
52            if index_type.pytype() in {"builtins.int", "builtins.slice"}:
53                return None
54            try:
55                index_type.getattr("__index__")
56                return None
57            except astroid.NotFoundError:
58                pass
59        elif isinstance(index_type, nodes.Slice):
60            # A slice can be present
61            # here after inferring the index node, which could
62            # be a `slice(...)` call for instance.
63            return self._check_invalid_slice_index(index_type)
64
65        # Anything else is an error
66        self.add_message("invalid-sequence-index", node=subscript)
67        return None
