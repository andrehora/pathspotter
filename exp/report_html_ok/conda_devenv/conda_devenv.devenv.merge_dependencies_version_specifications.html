41                # Consider the channel name as part of the package name.
42                # If multiple channels are specified, the package will be repeated.
43                package_name = m.group("package")
44                if m.group("channel"):
45                    package_name = m.group("channel") + package_name
46
47                package_version = m.group("version")
48
49            # OrderedDict is used as an ordered set, the value is ignored.
50            version_matchers = new_dependencies.setdefault(
51                package_name, collections.OrderedDict()
52            )
53            if len(package_version) > 0:
54                version_matchers[package_version] = True
55        else:
56            raise RuntimeError(f"Only strings and dicts are supported, got: {dep!r}")
57
58    result = set()
59    for dep_name, dep_version_matchers in new_dependencies.items():
60        if len(dep_version_matchers) > 0:
61            result.add(dep_name + " " + ",".join(dep_version_matchers))
62        else:
63            result.add(dep_name)
64
65    new_dict_dependencies = sorted(new_dict_dependencies, key=lambda x: list(x.keys()))
66    yaml_dict[key_to_merge] = sorted(result) + new_dict_dependencies
