36                # According to RFC5545 Section 3.3.10:
37                # https://tools.ietf.org/html/rfc5545#section-3.3.10
38                #
39                # > If the "DTSTART" property is specified as a date with UTC
40                # > time or a date with local time and time zone reference,
41                # > then the UNTIL rule part MUST be specified as a date with
42                # > UTC time.
43                raise ValueError(
44                    'RRULE UNTIL values must be specified in UTC when DTSTART '
45                    'is timezone-aware'
46                )
47
48        if count is not None and until:
49            warn("Using both 'count' and 'until' is inconsistent with RFC 5545"
50                 " and has been deprecated in dateutil. Future versions will "
51                 "raise an error.", DeprecationWarning)
52
53        if wkst is None:
54            self._wkst = calendar.firstweekday()
55        elif isinstance(wkst, integer_types):
56            self._wkst = wkst
57        else:
58            self._wkst = wkst.weekday
59
60        if bysetpos is None:
61            self._bysetpos = None
62        elif isinstance(bysetpos, integer_types):
63            if bysetpos == 0 or not (-366 <= bysetpos <= 366):
64                raise ValueError("bysetpos must be between 1 and 366, "
65                                 "or between -366 and -1")
66            self._bysetpos = (bysetpos,)
67        else:
68            self._bysetpos = tuple(bysetpos)
69            for pos in self._bysetpos:
70                if pos == 0 or not (-366 <= pos <= 366):
71                    raise ValueError("bysetpos must be between 1 and 366, "
72                                     "or between -366 and -1")
73
74        if self._bysetpos:
75            self._original_rule['bysetpos'] = self._bysetpos
76
77        if (byweekno is None and byyearday is None and bymonthday is None and
78                byweekday is None and byeaster is None):
79            if freq == YEARLY:
80                if bymonth is None:
81                    bymonth = dtstart.month
82                    self._original_rule['bymonth'] = None
83                bymonthday = dtstart.day
84                self._original_rule['bymonthday'] = None
85            elif freq == MONTHLY:
86                bymonthday = dtstart.day
87                self._original_rule['bymonthday'] = None
88            elif freq == WEEKLY:
89                byweekday = dtstart.weekday()
90                self._original_rule['byweekday'] = None
91
92        # bymonth
93        if bymonth is None:
94            self._bymonth = None
95        else:
96            if isinstance(bymonth, integer_types):
97                bymonth = (bymonth,)
98
99            self._bymonth = tuple(sorted(set(bymonth)))
100
101            if 'bymonth' not in self._original_rule:
102                self._original_rule['bymonth'] = self._bymonth
103
104        # byyearday
105        if byyearday is None:
106            self._byyearday = None
107        else:
108            if isinstance(byyearday, integer_types):
109                byyearday = (byyearday,)
110
111            self._byyearday = tuple(sorted(set(byyearday)))
112            self._original_rule['byyearday'] = self._byyearday
113
114        # byeaster
115        if byeaster is not None:
116            if not easter:
117                from dateutil import easter
118            if isinstance(byeaster, integer_types):
119                self._byeaster = (byeaster,)
120            else:
121                self._byeaster = tuple(sorted(byeaster))
122
123            self._original_rule['byeaster'] = self._byeaster
124        else:
125            self._byeaster = None
126
127        # bymonthday
128        if bymonthday is None:
129            self._bymonthday = ()
130            self._bynmonthday = ()
131        else:
132            if isinstance(bymonthday, integer_types):
133                bymonthday = (bymonthday,)
134
135            bymonthday = set(bymonthday)            # Ensure it's unique
136
137            self._bymonthday = tuple(sorted(x for x in bymonthday if x > 0))
138            self._bynmonthday = tuple(sorted(x for x in bymonthday if x < 0))
139
140            # Storing positive numbers first, then negative numbers
141            if 'bymonthday' not in self._original_rule:
142                self._original_rule['bymonthday'] = tuple(
143                    itertools.chain(self._bymonthday, self._bynmonthday))
144
145        # byweekno
146        if byweekno is None:
147            self._byweekno = None
148        else:
149            if isinstance(byweekno, integer_types):
150                byweekno = (byweekno,)
151
152            self._byweekno = tuple(sorted(set(byweekno)))
153
154            self._original_rule['byweekno'] = self._byweekno
155
156        # byweekday / bynweekday
157        if byweekday is None:
158            self._byweekday = None
159            self._bynweekday = None
160        else:
161            # If it's one of the valid non-sequence types, convert to a
162            # single-element sequence before the iterator that builds the
163            # byweekday set.
164            if isinstance(byweekday, integer_types) or hasattr(byweekday, "n"):
165                byweekday = (byweekday,)
166
167            self._byweekday = set()
168            self._bynweekday = set()
169            for wday in byweekday:
170                if isinstance(wday, integer_types):
171                    self._byweekday.add(wday)
172                elif not wday.n or freq > MONTHLY:
173                    self._byweekday.add(wday.weekday)
174                else:
175                    self._bynweekday.add((wday.weekday, wday.n))
176
177            if not self._byweekday:
178                self._byweekday = None
179            elif not self._bynweekday:
180                self._bynweekday = None
181
182            if self._byweekday is not None:
183                self._byweekday = tuple(sorted(self._byweekday))
184                orig_byweekday = [weekday(x) for x in self._byweekday]
185            else:
186                orig_byweekday = ()
187
188            if self._bynweekday is not None:
189                self._bynweekday = tuple(sorted(self._bynweekday))
190                orig_bynweekday = [weekday(*x) for x in self._bynweekday]
191            else:
192                orig_bynweekday = ()
193
194            if 'byweekday' not in self._original_rule:
195                self._original_rule['byweekday'] = tuple(itertools.chain(
196                    orig_byweekday, orig_bynweekday))
197
198        # byhour
199        if byhour is None:
200            if freq < HOURLY:
201                self._byhour = {dtstart.hour}
202            else:
203                self._byhour = None
204        else:
205            if isinstance(byhour, integer_types):
206                byhour = (byhour,)
207
208            if freq == HOURLY:
209                self._byhour = self.__construct_byset(start=dtstart.hour,
210                                                      byxxx=byhour,
211                                                      base=24)
212            else:
213                self._byhour = set(byhour)
214
215            self._byhour = tuple(sorted(self._byhour))
216            self._original_rule['byhour'] = self._byhour
217
218        # byminute
219        if byminute is None:
220            if freq < MINUTELY:
221                self._byminute = {dtstart.minute}
222            else:
223                self._byminute = None
224        else:
225            if isinstance(byminute, integer_types):
226                byminute = (byminute,)
227
228            if freq == MINUTELY:
229                self._byminute = self.__construct_byset(start=dtstart.minute,
230                                                        byxxx=byminute,
231                                                        base=60)
232            else:
233                self._byminute = set(byminute)
234
235            self._byminute = tuple(sorted(self._byminute))
236            self._original_rule['byminute'] = self._byminute
237
238        # bysecond
239        if bysecond is None:
240            if freq < SECONDLY:
241                self._bysecond = ((dtstart.second,))
242            else:
243                self._bysecond = None
244        else:
245            if isinstance(bysecond, integer_types):
246                bysecond = (bysecond,)
247
248            self._bysecond = set(bysecond)
249
250            if freq == SECONDLY:
251                self._bysecond = self.__construct_byset(start=dtstart.second,
252                                                        byxxx=bysecond,
253                                                        base=60)
254            else:
255                self._bysecond = set(bysecond)
256
257            self._bysecond = tuple(sorted(self._bysecond))
258            self._original_rule['bysecond'] = self._bysecond
259
260        if self._freq >= HOURLY:
261            self._timeset = None
262        else:
263            self._timeset = []
264            for hour in self._byhour:
265                for minute in self._byminute:
266                    for second in self._bysecond:
267                        self._timeset.append(
268                            datetime.time(hour, minute, second,
269                                          tzinfo=self._tzinfo))
270            self._timeset.sort()
271            self._timeset = tuple(self._timeset)
