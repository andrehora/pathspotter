2        """Returns the domain that should be set for the session cookie.
3
4        Uses ``SESSION_COOKIE_DOMAIN`` if it is configured, otherwise
5        falls back to detecting the domain based on ``SERVER_NAME``.
6
7        Once detected (or if not set at all), ``SESSION_COOKIE_DOMAIN`` is
8        updated to avoid re-running the logic.
9        """
10
11        rv = app.config["SESSION_COOKIE_DOMAIN"]
12
13        # set explicitly, or cached from SERVER_NAME detection
14        # if False, return None
15        if rv is not None:
16            return rv if rv else None
17
18        rv = app.config["SERVER_NAME"]
19
20        # server name not set, cache False to return none next time
21        if not rv:
22            app.config["SESSION_COOKIE_DOMAIN"] = False
23            return None
24
25        # chop off the port which is usually not supported by browsers
26        # remove any leading '.' since we'll add that later
27        rv = rv.rsplit(":", 1)[0].lstrip(".")
28
29        if "." not in rv:
30            # Chrome doesn't allow names without a '.'. This should only
31            # come up with localhost. Hack around this by not setting
32            # the name, and show a warning.
33            warnings.warn(
34                f"{rv!r} is not a valid cookie domain, it must contain"
35                " a '.'. Add an entry to your hosts file, for example"
36                f" '{rv}.localdomain', and use that instead."
37            )
38            app.config["SESSION_COOKIE_DOMAIN"] = False
39            return None
40
41        ip = is_ip(rv)
42
43        if ip:
44            warnings.warn(
45                "The session cookie domain is an IP address. This may not work"
46                " as intended in some browsers. Add an entry to your hosts"
47                ' file, for example "localhost.localdomain", and use that'
48                " instead."
49            )
50
51        # if this is not an ip and app is mounted at the root, allow subdomain
52        # matching by adding a '.' prefix
53        if self.get_cookie_path(app) == "/" and not ip:
54            rv = f".{rv}"
55
56        app.config["SESSION_COOKIE_DOMAIN"] = rv
57        return rv
