36        if call_site.has_invalid_arguments() or call_site.has_invalid_keywords():
37            # Can't make sense of this.
38            return
39
40        # Has the function signature changed in ways we cannot reliably detect?
41        if hasattr(called, "decorators") and decorated_with(
42            called, self.linter.config.signature_mutators
43        ):
44            return
45
46        num_positional_args = len(call_site.positional_arguments)
47        keyword_args = list(call_site.keyword_arguments.keys())
48        overload_function = is_overload_stub(called)
49
50        # Determine if we don't have a context for our call and we use variadics.
51        node_scope = node.scope()
52        if isinstance(node_scope, (nodes.Lambda, nodes.FunctionDef)):
53            has_no_context_positional_variadic = _no_context_variadic_positional(
54                node, node_scope
55            )
56            has_no_context_keywords_variadic = _no_context_variadic_keywords(
57                node, node_scope
58            )
59        else:
60            has_no_context_positional_variadic = (
61                has_no_context_keywords_variadic
62            ) = False
63
64        # These are coming from the functools.partial implementation in astroid
65        already_filled_positionals = getattr(called, "filled_positionals", 0)
66        already_filled_keywords = getattr(called, "filled_keywords", {})
67
68        keyword_args += list(already_filled_keywords)
69        num_positional_args += implicit_args + already_filled_positionals
70
71        # Decrement `num_positional_args` by 1 when a function call is assigned to a class attribute
72        # inside the class where the function is defined.
73        # This avoids emitting `too-many-function-args` since `num_positional_args`
74        # includes an implicit `self` argument which is not present in `called.args`.
75        if (
76            isinstance(node.frame(), nodes.ClassDef)
77            and isinstance(node.parent, (nodes.Assign, nodes.AnnAssign))
78            and isinstance(called, nodes.FunctionDef)
79            and called in node.frame().body
80            and num_positional_args > 0
81        ):
82            num_positional_args -= 1
83
84        # Analyze the list of formal parameters.
85        args = list(itertools.chain(called.args.posonlyargs or (), called.args.args))
86        num_mandatory_parameters = len(args) - len(called.args.defaults)
87        parameters: list[tuple[tuple[str | None, nodes.NodeNG | None], bool]] = []
88        parameter_name_to_index = {}
89        for i, arg in enumerate(args):
90            if isinstance(arg, nodes.Tuple):
91                name = None
92                # Don't store any parameter names within the tuple, since those
93                # are not assignable from keyword arguments.
94            else:
95                assert isinstance(arg, nodes.AssignName)
96                # This occurs with:
97                #    def f( (a), (b) ): pass
98                name = arg.name
99                parameter_name_to_index[name] = i
100            if i >= num_mandatory_parameters:
101                defval = called.args.defaults[i - num_mandatory_parameters]
102            else:
103                defval = None
104            parameters.append(((name, defval), False))
105
106        kwparams = {}
107        for i, arg in enumerate(called.args.kwonlyargs):
108            if isinstance(arg, nodes.Keyword):
109                name = arg.arg
110            else:
111                assert isinstance(arg, nodes.AssignName)
112                name = arg.name
113            kwparams[name] = [called.args.kw_defaults[i], False]
114
115        self._check_argument_order(
116            node, call_site, called, [p[0][0] for p in parameters]
117        )
118
119        # 1. Match the positional arguments.
120        for i in range(num_positional_args):
121            if i < len(parameters):
122                parameters[i] = (parameters[i][0], True)
123            elif called.args.vararg is not None:
124                # The remaining positional arguments get assigned to the *args
125                # parameter.
126                break
127            elif not overload_function:
128                # Too many positional arguments.
129                self.add_message(
130                    "too-many-function-args", node=node, args=(callable_name,)
131                )
132                break
133
134        # 2. Match the keyword arguments.
135        for keyword in keyword_args:
136            if keyword in parameter_name_to_index:
137                i = parameter_name_to_index[keyword]
138                if parameters[i][1]:
139                    # Duplicate definition of function parameter.
140
141                    # Might be too hard-coded, but this can actually
142                    # happen when using str.format and `self` is passed
143                    # by keyword argument, as in `.format(self=self)`.
144                    # It's perfectly valid to so, so we're just skipping
145                    # it if that's the case.
146                    if not (keyword == "self" and called.qname() in STR_FORMAT):
147                        self.add_message(
148                            "redundant-keyword-arg",
149                            node=node,
150                            args=(keyword, callable_name),
151                        )
152                else:
153                    parameters[i] = (parameters[i][0], True)
154            elif keyword in kwparams:
155                if kwparams[keyword][1]:
156                    # Duplicate definition of function parameter.
157                    self.add_message(
158                        "redundant-keyword-arg",
159                        node=node,
160                        args=(keyword, callable_name),
161                    )
162                else:
163                    kwparams[keyword][1] = True
164            elif called.args.kwarg is not None:
165                # The keyword argument gets assigned to the **kwargs parameter.
166                pass
167            elif isinstance(
168                called, nodes.FunctionDef
169            ) and self._keyword_argument_is_in_all_decorator_returns(called, keyword):
170                pass
171            elif not overload_function:
172                # Unexpected keyword argument.
173                self.add_message(
174                    "unexpected-keyword-arg", node=node, args=(keyword, callable_name)
175                )
176
177        # 3. Match the **kwargs, if any.
178        if node.kwargs:
179            for i, [(name, _defval), _assigned] in enumerate(parameters):
180                # Assume that *kwargs provides values for all remaining
181                # unassigned named parameters.
182                if name is not None:
183                    parameters[i] = (parameters[i][0], True)
184                else:
185                    # **kwargs can't assign to tuples.
186                    pass
187
188        # Check that any parameters without a default have been assigned
189        # values.
190        for [(name, defval), assigned] in parameters:
191            if (defval is None) and not assigned:
192                display_name = "<tuple>" if name is None else repr(name)
193                if not has_no_context_positional_variadic and not overload_function:
194                    self.add_message(
195                        "no-value-for-parameter",
196                        node=node,
197                        args=(display_name, callable_name),
198                    )
199
200        for name, val in kwparams.items():
201            defval, assigned = val
202            if (
203                defval is None
204                and not assigned
205                and not has_no_context_keywords_variadic
206                and not overload_function
207            ):
208                self.add_message(
209                    "missing-kwoa",
210                    node=node,
211                    args=(name, callable_name),
212                    confidence=INFERENCE,
213                )
