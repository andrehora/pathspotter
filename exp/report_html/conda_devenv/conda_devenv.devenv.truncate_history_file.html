10

12    "--ignore-history" or "--prune-hard" ar implemented we should revisit this function and
13    update the "conda-env" arguments.
14    """
15    if env_directory is None:
16        return  # Environment does not exists, no history to truncate
17
18    from os.path import isfile, join
19    from time import time
20    from shutil import copyfile
21
22    history_filename = join(env_directory, "conda-meta", "history")
23    history_backup_filename = "{}.{}".format(history_filename, time())
